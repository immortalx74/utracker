START_HOOK_SIG(default_start){
    named_maps = named_maps_values;
    named_map_count = ArrayCount(named_maps_values);
    
    default_4coder_initialize(app);
    // default_4coder_side_by_side_panels(app, files, file_count);
    default_4coder_one_panel(app);
    get_or_open_build_panel(app);
    exec_command(app, open_panel_vsplit);
    View_Summary special_view = get_view(app, build_footer_panel_view_id, AccessAll);
    view_set_split_proportion( app, &special_view, 0.02f );

    if (global_config.automatically_load_project){
        load_project(app);
    }
    
    // no meaning for return
    return(0);
}

CUSTOM_COMMAND_SIG(toggle_build_panel)
CUSTOM_DOC("Toggles the build panel")
{
    static bool toggler = true;
    
    if (toggler)
    {
        // get_or_open_build_panel(app);
	    View_Summary special_view = get_view(app, build_footer_panel_view_id, AccessAll);
	    view_set_split_proportion( app, &special_view, 0.5f );
    }
    else
    {
        // close_build_footer_panel(app);
	    View_Summary special_view = get_view(app, build_footer_panel_view_id, AccessAll);
	    view_set_split_proportion( app, &special_view, 0.05f );
    }
    
    toggler = !toggler;
}
